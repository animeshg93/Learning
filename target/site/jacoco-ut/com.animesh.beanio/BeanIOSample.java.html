package com.animesh.beanio;

import java.io.StringWriter;

import org.beanio.BeanWriter;
import org.beanio.StreamFactory;
import org.beanio.builder.FixedLengthParserBuilder;
import org.beanio.builder.StreamBuilder;

public class BeanIOSample {
	public void readCSVFileUsingBeanIo() {
		StreamFactory factory = StreamFactory.newInstance();
		StringWriter writer = new StringWriter();
		StreamBuilder builder = new StreamBuilder("person").format("fixedlength")
				.parser(new FixedLengthParserBuilder().recordTerminator("\n")).addRecord(Contact.class)
				.ignoreUnidentifiedRecords();
		factory.define(builder);

		BeanWriter out = factory.createWriter("person", writer);
		Contact c = makeContact();
		out.write(c);

		out.flush();
		out.close();
	}

	private Contact makeContact() {
		Contact c = new Contact();
		c.setCity("Alabama");
		c.setFirstName("Animesh");
		c.setLastName("Gandhi");
		c.setState("GA");
		c.setStreet("Main St");
		c.setZip("8524");
		return c;
	}

	public static void main(String[] args) {
		BeanIOSample sample = new BeanIOSample();
		long start = System.currentTimeMillis();
		sample.readCSVFileUsingBeanIo();
		long end = System.currentTimeMillis();
		System.out.println("TIME TAKEN >>> " + (end - start));
	}
}
